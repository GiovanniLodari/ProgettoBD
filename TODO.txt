SELECT id_str AS Id, user.name AS utente, favorite_count AS MiPiace, reply_count AS NumeroRisposte, retweet_count as NumeroRetweet, quote_count AS NumeroCitazioni, (favorite_count + reply_count + retweet_count + quote_count) AS InterazioniTotali
FROM disasters 
ORDER BY InterazioniTotali DESC

SELECT retweeted_status.user.name, retweeted_status.user.screen_name, MAX(retweeted_status.user.followers_count) as Followers,  MAX(retweeted_status.user.friends_count) as Friends, Count(retweeted_status.id_str) as NumeroTweet, SUM(retweeted_status.favorite_count) as Mi_Piace_Totali
FROM disasters
GROUP by  retweeted_status.user.name, retweeted_status.user.screen_name
ORDER BY  Mi_Piace_Totali DESC
